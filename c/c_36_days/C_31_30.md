```c
  //数组指针，以数组为单位的指针变量，此刻_countof(失效）

  int i = -1;

  {
    int j = -1;
```

```c
// AddHead AddTail Print Remove 

#include <stdio.h>
#include <stdlib.h>

typedef double DATA;

struct SNode // 元素节点
{
  DATA data;
  struct SNode* pNext;
};
struct SNode* g_pHead = NULL;

// 头插法
void AddHead (DATA d)
{
  struct SNode* p = (struct SNode*)malloc(sizeof(SNode));
  if (!p)
    return;
  p->data = d;
  p->pNext = g_pHead;
  g_pHead = p;
}
// 尾插法
void AddTail (DATA d)
{
  struct SNode* p = (struct SNode*)malloc(sizeof(SNode));
  if (!p)
    return;
  p->data = d;
  p->pNext = NULL;
  
  if (g_pHead == NULL)
    g_pHead = p, return;
  
  struct SNode* pTail = g_pHead; // 找尾巴
  while (pTail->pNext) // 我产生过疑问pTail不是NULL？ 因为pTail是最后一个元素时 不会进入判断
    pTail = pTail->pNext; // 链表没有p++
  pTail->pNext = p;
}
// 遍历
void Print ()
{
  int sum = 0;
  struct SNode* p = g_pHead;
  while (p)
  {
	printf("%-6.2f", p->data); // %-6.2f -左对齐  6个字符宽 2位小数保留
    p = p->pNext;
    sum++;
  }
  printf("总共有 %d 条数据 \n", sum);
}
// 删除
int Remove (DATA d)
{
  struct SNode* p = g_pHead, *q = NULL;
  while (p)
  {
    if (p->data == d)
    {
		if (p == g_pHead)
          	g_pHead = p->pNext;
      	else 
          q->pNext = p->pNext;
      
      	free(p);
      	return 1;
    }
    q = p;
    p = p->pNext;
  }
  return 0;
}

int main ()
{
	int d = 6;
  	Print();
  	while (--d > 0)
      Addhead(d * 10);
  	Print();
  	Remove(30);
  	Print();
  	return 0;
}
```

### 2. select-sort

```c
// 下标法的选择排序 
// 链表只能使用 归并排序
// 思路：找出最小值 或者 最大值 与第一个或者最后一个进行交换
#include <stdio.h>
#include <stdlib.h>
int a[] = { 78,108,55,32,41,69,11,35,18,21 };

// 下标法-选择排序
void SelectSort (int* p, int nLen)
{
  int i = -1;
  while (++i < nLen-1)
  {
    int min = i, j = i;
    while (++j < nLen)
    {
      if (p[j] < p[min])
        min = j;
    }
    
    if (min != i)
    {
      int t = p[min];
      p[min] = p[i];
      p[i] = t;
    }
  }
}
// 指针法-选择排序
void SelectSort (int* pData, int nLen)
{
  int *p = pData;
  while (p - pData < nLen-1) // 这个指针遍历一定要熟悉
  {
    int *q = p, *m = p;
    while (++q - pData < nLen)
    {
      if (*q < *m)
        m = q;
    }
    
    if (m != p)
    {
      int t = *p;
      *p = *m;
      *m = t;
    }
    ++p;
  }
}
// 遍历
void Print (const int* p, int nLen)
{
  while (nLen--)
    printf("%d ", *p++);
  putchar(10);
}
int main ()
{
  Print(a, _countof(a));
  SelectSort(a, _countof(a));
  Print(a, _countof(a));
  return 0;
}
```
