### 2. malloc 堆

```c
#define  _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main ()
{
  int n = 0;
  printf("请输入你需要多少个元素的数组：");
  scanf_s("%d", &n);
  char* p = (char *)malloc(n); // 类比：char [n]数组
  
  strcpy(p, "所以计算出来的值比字符串本身长度多");
  puts(p);
  free(p); // 释放堆内存
}
```
