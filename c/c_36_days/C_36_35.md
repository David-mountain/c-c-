### 2. union vs struct

```c
#include <stdio.h>
struct SData 
{
	short s;
  	int n;
  	double d;
}
union UData
{
	short s;
	int n;
	double d;
}

int main ()
{
  SData d1;
  UData d2;
  printf("0x%p 0x%p 0x%p \n", &d1.s, &d1.n, &d1.d);
  printf("0x%p 0x%p 0x%p \n", &d2.s, &d2.n, &d2.d);
  printf("sizeof(SData)=%lld \n", sizeof(struct SData));
  printf("sizeof(UData)=%lld \n", sizeof(union UData));
  
  return 0;
}
```

```c
#include <stdio.h>

void PrintBinary (int n)
{
  int i = 32;
  while (i--)
  {
    putchar((n >> i & 1) + '0'); // 1 & 1 = 1     0 & 1 = 0
    if (i % 4 == 0)
      putchar(32);
  }
  putchar(10);
}
void Menu ()
{
  puts("1、在某个位置放入1");
  puts("2、在某个位置放入0");
  puts("3、判断某个位置是不是1");
  puts("0、退出");
  int i = 0;
  scanf_s("%d", &i);
  return i;
}

int main ()
{
  int n = 0x7A8549E3;
  int i = 0;
  int x = 0;

  {
    PrintBinary(n);
    i = Menu();

        {
          puts("请问你要在第几位放1？");
```

### 8. calculator-command

```c
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

// dos参数 主函数参数，命令行参数
// int main (int argc, char* argv[])
int main (int argc, char** argv)
{
  if (argv < 2)
  {
    puts("【支持+-*/】命令格式：calc 88.88-66.9");
    return -1;
  }
  
  char s[256];
  s[0] = 0;
  while (*++argv)
    strcat(s, *argv);
  char* p = strpbrk(s, "+-*/");
  if (!p)
  {
    puts("【支持+-*/】命令格式：calc 88.88-66.9");
    return -1;
  }
  char c = *p;
  *p++ = '\0';
  double left = atof(s),
  		right = atof(p);
  		result = 0.0;
  switch (c)
  {
    case '+':
      result = left + right;
      break;
    case '-':
      result = left - right;
      break;
    case '*':
      result = left * right;
      break;
    case '/':
      result = left / right;
      break;
  }
  printf("%g %c %g = %g \n", left, c, right, result);
  
  return 0;
}

// 执行 .exe可执行文件  
// ./exe 6-2
```

```c
#include <stdio.h>  // printf
#include <stdlib.h> // qsort

int values[] = { 40, 10, 100, 90, 20, 25 };
int compare (const void* a, const void* b)
{
  return (*(int*)a - *(int*)b);
}

int main ()
{
  int n{0};
  qsort(values, _countof(values), sizeof(*values), compare);
  for (n = 0; n < _countof(values); ++n)
    printf("%d ", values[n]);
  
  return 0;
}
```
