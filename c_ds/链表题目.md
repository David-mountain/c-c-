### 顺序表反转
    void ReverSeList (SequenList* pList)
    {
        int temp;
        int count, i;
        if ( pList->last == 0 || pList->last == 1 )
        {
            return;
        }
        
        count = pList->last / 2;
        for ( i=0; i<count; i++ )
        {
            /0 + (last-1) = last-1 
            /1 + (last-2) = last-1
            /i + ? = last-1
            /? = last-1-i; 对调后面的元素下标

            temp = pList->data[i];
            pList->data[i] = pList->data[pList->last-1-i];
            pList->data[pList->last-1-i] = temp;
        }
    }
