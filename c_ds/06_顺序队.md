# 顺序队

## 原始模板

```c
#include <stdio.h>
#include <stdlib.h>

#define QUEUE_SIZE 100
#define TRUE 1
#define FALSE 0
typedef int data dataType;
typedef int BOOL;
typedef struct queue
{
  dataType data[QUEUE_SIZE];
  int front, rear;
}SeqQueue;

// ... 后续代码放main之上
int main ()
{
  
}
```

## 初始化

```c
void Init_SqQueue (SeqQueue *sq)
{
  sq->front = QUEUE_SIZE - 1;
  sq->rear = QUEUE_SIZE - 1;
}
```

## 判空

```c
void IsEmpty_SqQueue (SeqQueue *sq)
{
  if ( sq->rear == sq->front )
    	return TRUE;
  return FALSE;
}
```

## 获取

```c
// 返回数组下标
int Get_SqQueue (SeqQueue *sq)
{
  if ( !IsEmpty_SqQueue(sq) )
  {
    return (sq->front + 1) % QUEUE_SIZE;
  }
  return -1;
}
```

## 入队

```c
BOOL EnSqQueue (SeqQueue *sq, dataType x)
{
  if ( sq->front == (sq->rear+1)%QUEUE_SIZE )
  	return FALSE;
  sq->rear = (sq->rear+1)%QUEUE_SIZE;
  sq->data[sq->rear] = x;
  return TRUE;
}
```

## 出队

```c
int DeSqQueue (SeqQueue *sq)
{
  if ( !IsEmpty_SqQueue(sq) )
  {
	sq->front = (sq->front+1)%QUEUE_SIZE;
    return sq->front;
  }
  return -1;
}
```
